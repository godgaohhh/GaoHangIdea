实现计算当前为校历第几周的代码

需要设置上半学期开始时间，以及下半学期开始时间

import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.Date;

/**
 * @author gyh
 */
public class cs {
    public static void main(String[] args) {
        Calendar calendar = Calendar.getInstance();
        calendar.setFirstDayOfWeek(Calendar.MONDAY);
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd");
        if(calendar.get(Calendar.MONTH)+1 >= 8 || calendar.get(Calendar.MONTH)+1 < 2 ){
            String startTime = "2022-08-30";
            try {
                Date stD = simpleDateFormat.parse(startTime);
                long st = stD.getTime();
                int year = calendar.get(Calendar.YEAR);
                int weekFoYear1 = calendar.get(Calendar.WEEK_OF_YEAR);
                System.out.println("当前为2022年的第"+weekFoYear1+ "周");
                calendar.setTimeInMillis(st);
                int weekFoYear2 = calendar.get(Calendar.WEEK_OF_YEAR);
                System.out.println("当前为"+year+"年下学期的第"+(weekFoYear1 - weekFoYear2 + 1)+"周");
            } catch (ParseException e) {
                e.printStackTrace();
            }
        }
        if(calendar.get(Calendar.MONTH)+1 >= 2 && calendar.get(Calendar.MONTH)+1 < 8){
            String startTime = "2022-02-28";
            try {
                Date stD = simpleDateFormat.parse(startTime);
                long st = stD.getTime();
                int year = calendar.get(Calendar.YEAR);
                int weekFoYear1 = calendar.get(Calendar.WEEK_OF_YEAR);
                System.out.println("当前为"+year+"年的第"+weekFoYear1+ "周");
                calendar.setTimeInMillis(st);
                int weekFoYear2 = calendar.get(Calendar.WEEK_OF_YEAR);
                System.out.println("当前为"+year+"年上学期的第"+(weekFoYear1 - weekFoYear2 + 1)+"周");
            } catch (ParseException e) {
                e.printStackTrace();
            }
        }
    }
}

<template>
    <div>

        <button @click="isShowForm = true">添加</button>
        <hr>
        <!-- 表格 -->
        <v-list @del="del" :menus="menus"></v-list>

    </div>

<script>
// 组件引入
import vList from "./list.vue"
import vForm from "./form.vue"
// 对外暴露
export default {
    // 组件注册
    components: {
        vList, vForm
    },
    data() {
        return {
            isShowForm: false,
            menus: [],//新闻的集合
        }
    },
    methods:{
        add(arg){
            // console.log(arg,'index')
            this.menus.push(arg.obj)
            this.isShowForm = false;
        },
        hideMaskForm(arg){
            // console.log(arg)
            this.isShowForm = arg;
        },
        del(arg){
            // console.log(arg,"index")
            this.menus.splice(arg.index,1)
        }
    }
}
</script>

    data() {
        return {
            title: "",
            content: ""

        // 取消

            this.$emit("hideMaskForm",false);
        }
    }
}
</script>

说明：守卫分类

  1.全局前置守卫   --  整个园区的门卫

1. 全局前置守卫-【实现登录拦截】

说明：

router.beforeEach(to,from,next)
to： 即将进入的路由对象
from: 即将离开的路由对象
next:向下运行
需求： 登录之后，进入项目，否则进入登录页面；

1.login.vue设计一个标识，用来判断是否登录

   methods: {
        login() {
            // console.log(123)
            // 登录的业务逻辑
            var user = "189", pass = 123; //假的
            var { phone, password } = this; //用户填写的
            if(user != phone &&  pass != password){
                alert("用户名密码错误");
                return;
            }
            // 登录成功
            // 存储登录状态
            // 本地存储
            localStorage.setItem("loginStatus",true)
            // 认证成功

    }

2.router/index.js 设置全局前置守卫

/ 路由前置守卫
// 需求： 不登录不让正常访问到项目，直接进入登录页面
router.beforeEach((to, from, next) => {
  // console.log(to)
  if(to.path =="/login"){
    next()
    return;
  }
  // to   去哪
  // from  从哪来的
  // next()  向下运行
  // 判断是否登录，登录则next，没登录则 跳转到登录页面
  var loginStatus = localStorage.getItem('loginStatus') || false;
  // console.log(loginStatus)
  if (loginStatus) {
    // 证明登陆了
    next();
  } else {
    // console.log("未登录")
    // 跳转到登录页面
    next("/login")
  }

})

解决跨域方式：

1.将前端后端部署到同一个服务器下

3.前端使用webpack设置代理

4.1.项目下创建 vue.config.js

语法:

publicPath:""  打包后的基础路径
indexPath:""   打包后的入口文件
outputDir:""   打包后的文件夹
assetsDir:""   打包后的静态资源
devServe:{proxy:{}} 配置代理

代码：vue.config.js

// webpack配置文件
module.exports = {
    // 1.打包之后的入口文件
    indexPath:"index.html",
    // 2.打包之后的路径
    publicPath:"",
    // 3.打包之后的包名
    outputDir:"dist",
    // 4.打包之后静态资源目录
    assetsDir:"static",
    // 5.设置代理
    devServer:{
        proxy:{
            "/api":{
                target:"http://localhost:3000",
                changeOrigin:true, //允许跨域
                ws:true,
            },
        }
    }
}
